\subsection{Initial Conditions}
\begin{frame}{Initial Conditions - Part 1}
	\begin{itemize}
	\item Position of the black hole, \(\vec{r}_{BH} = \langle 0, 0, 0 \rangle\ \mathrm{m}\)
	\item Position of the star, \(\vec{r}_{star}\) given to us in a Python List at given time intervals.
    \item Mass of the star, \(m_{star} = 2\times 10^{30}\ \mathrm{kg}\)
    \item \(\Delta t = 86400\ \mathrm{s}\)
	\end{itemize}
\end{frame}

\begin{frame}{Initial Conditions - Part 2}
	\begin{itemize}
	\item Position of the black hole, \(\vec{r}_{BH} = \langle 0, 0, 0 \rangle\ \mathrm{m}\)
    \item Mass of the blackhole, \(m_{BH} = 1.35641\times 10^{34}\ \mathrm{kg}\) (from part 1)
	\item Initial position of the planet, \(\vec{r}_{planet} = \langle 1\times 10^{10}, 0, 0 \rangle\ \mathrm{m}\)
    \item Radius of the planet, \(a = 1\times 10^{8}\ \mathrm{m}\)
    \item Initial position of the ship, \(\vec{r}_{ship} = \langle 1\times 10^{10}, 1\times 10^{8}, 0 \rangle\ \mathrm{m}\)
	\end{itemize}
\end{frame}

\begin{frame}{System and Surroundings - Part 1}
	\begin{itemize}
	\item \textbf{System:} The star (hereafter referred to `object')
	\item \textbf{Surroundings:} Everything else (black hole etc.)
	\end{itemize}
\end{frame}

\begin{frame}{System and Surroundings - Part 2}
	\begin{itemize}
	\item \textbf{System:} The ship (hereafter referred to `object')
	\item \textbf{Surroundings:} Everything else (black hole, planet etc.)
	\end{itemize}
\end{frame}
